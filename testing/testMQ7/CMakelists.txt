cmake_minimum_required(VERSION 3.13)

include(pico_sdk_import.cmake)

project(mq7_testing_project C CXX ASM)
set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 17)

pico_sdk_init()

add_executable(mq7_test
    main.cpp
)

# Abilita output su USB
pico_enable_stdio_usb(mq7_test 1)
pico_enable_stdio_uart(mq7_test 0)

# Link alle librerie hardware
target_link_libraries(mq7_test 
    pico_stdlib 
    hardware_adc
    hardware_pwm
)

pico_add_extra_outputs(mq7_test)
